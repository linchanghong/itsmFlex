            import com.itsm.common.utils.AppCore;
            import com.itsm.serviceManager.module.bugManager.code.BugUtil;
            import com.itsm.serviceManager.common.mxml.BugFieldSelect;

            import mx.rpc.events.ResultEvent;
            
            private var appCore:AppCore = AppCore.getInstance();
            
            private var selectComponent:BugFieldSelect;//主程序的引用
            
            [Bindable]private var selectName:String = ""; //选中名字
            private var selectData:ArrayCollection = new ArrayCollection();    //选中数据
            [Bindable]private var gdData:Array = null;
            //3：代表查询需求
            private static const FLAG:int = 3;
            
            /**
             * 绑定树数据
             */
            protected function tree_itemClickHandler(event:ListEvent):void
            {
                gdData = ["demandManagerAPI", "DemandManagerAPI", 'initHomePageFast4SelectApply', [FLAG]];
            }
            
            /**
             * 确定按钮事件监听 
             */
            protected function sureClickHandler(event:Event):void
            {
                
                //给FieldSelect中的控件赋值
                //                selectComponent.selectObj = dg.selectedItem;
                //                selectComponent.selText.text = dg.selectedItem.applyTitle;
                if(selectData != null && selectData.length > 0){
                    if(selectComponent.isMultipleSelected){    
                        selectComponent.selectObj = selectData;
                    }else
                        selectComponent.selectObj = selectData.getItemAt(0);
                    
                    selectComponent.sText = selectComponent.selText.text = selectName;
                    
                    PopUpManager.removePopUp(this);
                }else{
                    TAlert.show("请选择需求单！","温馨提示");
                }
                
            }
            
            /**
             * 关闭窗口
             */
            private function closeWindow(event:MouseEvent):void
            {
                PopUpManager.removePopUp(this);
            }
            
            /**
             * 查询按钮事件监听 
             */
            protected function queryClikHandler(event:Event):void
            {
                //                var item:XML = tree.selectedItem as XML;
                //                if(item == null){
                //                    TAlert.show("请选择公司或者部门！");
                //                    return;
                //                }
                
                gdData = ["selectUserAPI", "SelectUserAPI", "queryUserAppliesByLike", 
                    [3, 2, applyTitle.text]];
            }
            
            protected function creationCompleteHandler(event:FlexEvent):void
            {
                selectComponent = this.owner as BugFieldSelect; //FieldSelect的引用
                tree_itemClickHandler(null);
            }
            
            protected function dg_itemClickHandler(event:ListEvent):void
            {
                var obj:Object = dg.selectedItem;
                
                if(obj != null) {
                    if(selectComponent.isMultipleSelected){
                        
                        if(selectName!=""){
                            var isHave:Boolean = true;
                            var names:Array = selectName.split(",");
                            for(var i:int=0; i<names.length; i++){
                                if(obj.applyTitle==names[i]){
                                    names.splice(i, 1);
                                    selectData.removeItemAt(i);
                                    isHave=false;
                                    break;
                                }
                            }
                            
                            if(isHave) {
                                names.push(obj.applyTitle);
                                selectData.addItem(obj);
                            }
                            
                            selectName=names.join(",");
                        }else{
                            
                            selectName = obj.applyTitle;
                            selectData.addItem(obj);
                        }
                    }else{
                        if(obj.applyStatus > 1 && obj.applyStatus < 3){
                            BugUtil.handle_server_method("demandManagerAPI","DemandManagerAPI",
                                "getUserIdByPersonId",
                                [selectComponent.parame], function(event:ResultEvent):void{
                                    var resultObj:Object = BugUtil.getResultObj(event);
                                    
                                    if(resultObj == obj.replyer.id){
                                        applyTitle.setStyle("color","black");
                                        selectName = obj.applyTitle;
                                        selectData.removeAll();
                                        selectData.addItem(obj);
                                    }else{
                                        applyTitle.setStyle("color","red");
                                        selectName = "请联系该请求单负责人！";
                                    }
                                });
                        }else{
                            applyTitle.setStyle("color","red");
                            selectName = "该请求单未进入流程或流程结束！";
                        }
                    }
                }
            }
